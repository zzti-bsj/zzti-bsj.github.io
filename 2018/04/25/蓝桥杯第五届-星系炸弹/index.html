星系炸弹：(关于时间 日期 的题目)

  • 对于时间日期的题目及源码编写比较少见，偏门，所以就copy了一下此题的源码以待以后套用。
  • 闰年的定义
  • 月份的处理方法
  • while循环中也是重点
  • 求未来某一个日期的方法
    代码
#include<iostream>
#include<cstdio>
using namespace std;
//判断闰年的函数
int r(int y)
{
    if(y%100!=0&&y%4==0||y%400==0)
        return 1;
    else
        return 0;
}
int mon[12]={31,28,31,30,31,30,31,31,30,31,30,31};//月份的定义。
//初始化年月日
int year=2014;
int month=11;
int day=9;
int num=1000;
int main()
{
    while(num>0)
    {
        if(mon[month-1]-day<num)
        {
            num-=(mon[month-1]-day+1);//直接进入到下一个月的第一天。
            day=1;
            month++;
            if(month>12)
            {
                year++;
                month=1;
                if(r(year))
                    mon[1]=29;
                else
                    mon[1]=28;
            }
        }
        else
        {
            day+=num;
            num=0;
        }
    }
    cout<<year<<' '<<month<<' '<<day<<endl;
    return 0;
}
