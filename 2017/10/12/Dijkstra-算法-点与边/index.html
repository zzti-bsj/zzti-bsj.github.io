Dijkstra 算法 点与边

Dijkstra:
单元最短路径算法。
图：点—-点
重点：点与边的关系。

Dijkstra算法源码：

#include<stdio.h>
int main()
{
    int e[10][10];
    int dis[10];
    int book[10];
    int n, m, t1, t2, t3, u, v, min;
    int inf = 99999999;
    //读入n和m，n表示顶点数，m表示边数。
    //初始化。
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n;i++)
    for (int j = 1; j <= n;j++)
    if (i == j) e[i][j] = 0;
    else
        e[i][j] = inf;
    //读入边。
    for (int i = 1; i <= m; i++)
    {
        scanf("%d%d%d", &t1, &t2, &t3);
        e[t1][t2] = t3;
    }
    //初始化dis数组，这里是1到其他顶点的初始路程。
    //book初始化。
    for (int i = 1; i <= n; i++)
    {
        dis[i] = e[1][i];
        book[i] = 0;
    }
    book[1] = 1;
    //Dijkstra算法核心。
    for (int i = 1; i <= n - 1; i++)
    {
        min = inf;
        for (int j = 1; j <= n; j++)
        {
            if (!book[j] && dis[j] < min)
            {
                min = dis[j];
                u = j;
            }
        }
        book[u] = 1;
        for (v = 1; v <= n; v++)
        {
            if (e[u][v] < inf)
            {
                if (dis[v]>e[u][v] + dis[u])
                    dis[v] = e[u][v] + dis[u];
            }
        }
    }
    for (int i = 1; i <= n; i++)
        printf("%d ", dis[i]);
    return 0;
}
