os模块

常用方法

os: Operating System
  想一想我们能想到的os能为我们做的事，并且是常做的事情。

  1. 创建文件夹
  2. 创建多重嵌套文件夹
  3. 修改文件名称
  4. 删除文件(只能删除空文件夹)
  5. 删除文件夹(只能删除空文件夹)
  6. 获取文件路径
  7. 获取文件名以及后缀名
  8. 获取文件的后缀名和文件路径
  9. 获取文件信息
    os模块的这些东西大多都与文件夹/文件有关系。一想到文件及文件夹问题，我们需要来看看os模块和os的path模块了

    os模块的一些常用用法

import os
# 0、模块函数cpu_count() 输出cpu的数量
print(os.cpu_count())

# 1、模块函数makedirs() 创建文件夹
# makedirs和mkdir()在makedirs()的函数说明里说的明明白白的，makedirs可以创建多级目录，如果目录存在的话，并且可以设置参数exists_ok=True，不会报错。
os.mkdir("image")
os.makedirs("image/a/b/c")

# 2、模块函数getcwd() 获得当前工作目录的绝对路径
print(os.getcwd())

# 3、模块函数listdir() 打印当前路径下的文件/文件夹的名称，不会再深入下一层目录
# 无参，返回当前路径下的文件/文件夹名称
# 有参，返回参数路径下的文件/文件夹名称
print(os.listdir("D:/python_project/python_student_zhiyou"))

# 4、模块函数walk() 返回指定的根目录(包括参数根目录)的文件夹中所有文件夹对应的root, dirs, files
gen_path = os.walk("D:\python_project\python_student_zhiyou")
for root, dirs, files in gen_path:
    print(root, "***", dirs, "***", files)

# 5、模块函数remove() 删除指定文件，如果想删除某个路径下的所有文件，可以配合walk使用，删除某个文件夹下的所有文件
os.remove(r"D:/python_project/python_student_zhiyou/imges/favicon.png")

# 6、path.splitext() 以元组形式获取文件名/后缀名(filename, 后缀名)
gen_path = os.walk("D:\python_project\python_student_zhiyou")
for root, dirs, files in gen_path:
    print(root, "***", dirs, "***", files)
    for file in files:
        # 由此可见，splitext返回的是一个文件名和其后缀名
        print(os.path.splitext(file))

# 7、模块函数replace() 给文件重命名
os.replace(r"D:\python_project\python_student_zhiyou\imges\test\favicon.png", r"D:\python_project\python_student_zhiyou\imges\test\t.png")

# 8、模块函数rmdir()/removedirs() 删除文件夹
# os.rmdir() # 删除一个空文件夹
os.rmdir('a')
print(os.getcwd())

# 9、os.removedirs() # 删除多重嵌套的文件夹（空文件夹）
os.removedirs(r'b\c\e')
print(os.getcwd())

# 10、查看某文件的信息
file_stas = os.stat('1.txt')

os之path模块常用用法

# 1、判断是否是绝对路径
print(os.path.isabs('1.txt'))

# 2、返回文件的路径、文件的扩展名
path, ext_name = os.path.splitext('1.txt')
print(path, ext_name)

# 3、返回上一级的路径
path = os.path.dirname('/iamge/2.jpg')
print(path)

# 4、返回路径的最后的文件夹或者文件名称
path = os.path.basename('/iamge/2.jpg')
print(path)

# 5、获取文件的信息
# os.path.getatime()/os.path.getctime()/os.path.getmtime()/os.path.getsize()等这些方法

删除文件夹下的多级文件夹及文件 

import os

path_root = r'D:\python_project\python_student_zhiyou\2019-7-29\t'

def dir_path(path_root):
    genpath = os.walk(path_root)
    for path in genpath:
        print(path)
        """
        先判断是不是文件夹
        1、如果是文件夹，传入一个路径，调用自己
        1、如果是文件，删除文件及path_root。
        """
        if path[1]:
            for dir in path[1]:
                tem_path = os.path.join(path[0], dir)
                dir_path(tem_path)
        for filename in path[2]:
            tem_path = os.path.join(path[0], filename)
            os.remove(tem_path)
            # 删除这个文件夹中的文件之后删除这个文件夹
            os.removedirs(path_root)


# 调用这个函数
dir_path(path_root)
