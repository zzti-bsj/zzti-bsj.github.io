mysql分区分表

  • 2020-03-01
  • Bsj

为什么分区分表？

分表：提高表的查询性能（查询速度）
分区：提高数据库的并发性能
既分表又分区：提高表的查询速度以及提高数据库的并发性能

分表

  选择恰当的分表策略，按照一定的规则分解成多张具有独立存储空间的实体表

特点：

  1. 解决了单表数据量大带来的查询效率下降的问题（提高了表的访问速度）
  2. 无法给数据库并发处理能力带来质的提升

子表可以在同一磁盘，也可以在不同机器上。

分库

  对数据库进行拆分，将数据分段划分在多个位置存放，从而提高数据库写入能力

特点：

  1. 提高数据库的并发性能

既分库又分表

  分成不同的数据库，在不同的数据库中又有多个相同的表结构，用来存储结构相同的数据。
解决的问题：

  1. 数据库面临的高并发压力
  2. 海量数据的存储问题

分库分表策略

  1. 中间变量 = user_id % (分库数量 * 每个库的表数量)
  2. 库 = 取整数 (中间变量 / 每个库的表数量)
  3. 表 = 中间变量 % 每个库的表数量

import sqlite3
import pymysql
from DB_tool import DBTool,DB_type
"""
存储stu信息：name，id

1、使用sqlite，创建256个数据库。
每个数据集创建20个表

2、插入一条stu信息
id = 3000
name = '张三'

3、查询这条信息在哪个数据库哪个表中

# 分库分表策略
# temp = temp % (table_num * db_num)
# db_id = temp / table_num
# table_id = temp % table_num
"""

id = 3000
name = '张三'
temp = id % (256 * 20)
db_id = temp // 20
tab_id = temp % 20
print(db_id, tab_id)


def con_sqlite(db_name):
    db = sqlite3.connect(db_name)
    return db


# 创建多个数据库及表
def multi_sqlite():
    # 创建256个库
    for i in range(256):
        # 连接256个数据库
        db = con_sqlite(f"db_{i}")
        cursor = db.cursor()
        # 每个数据库创建256个表
        print(f'创建第{i}数据库')
        for j in range(20):
            sql = f"""CREATE TABLE table_{j}(name VARCHAR (25), age INTEGER PRIMARY KEY AUTOINCREMENT);"""
            cursor.execute(sql)

# 按照分库分表策略插入数据到特定的数据库中
def insert(db_id, tab_id, id, name):
    db = con_sqlite(f'db_{db_id}')
    cursor = db.cursor()
    sql = f"""
        INSERT INTO table_{tab_id} VALUES (?,?);
    """
    cursor.execute(sql, (name, id))
    db.commit()


# 查询数据在哪个数据库中
def select(db_id, tab_id):
    db = con_sqlite(f'db_{db_id}')
    cursor = db.cursor()
    sql = f"""
        SELECT * FROM table_{tab_id};
    """
    cursor.execute(sql)
    result = cursor.fetchall()
    print(result)

DB_tool代码

# coding:utf-8
import sqlite3
# db=DataBase=数据库
import pymysql
from enum import Enum

MYSQL_COF = {
    # host主机名 user数据库的用户名 password数据库登录密码 database 要连接的数据库
    'host': '127.0.0.1',
    'port': 3306,
    'user': 'root',
    'password': 'root',
    'database': 'db',
    'charset': 'utf8'
}
SQLITE_COF = {
    'database': 'db.sqlite'
}


class DB_type(Enum):
    MYSQL = 'mysql'
    SQLITE = 'sqlite'


class DBTool:

    def __init__(self, db_type: DB_type = DB_type.MYSQL):

        # 提示用户输入正确的数据库类型
        if not isinstance(db_type, DB_type):
            raise TypeError("参数类型不对,请使用'DB_type.MYSQL'的形式")
        if db_type.value == 'mysql':
            con = pymysql.connect(**MYSQL_COF)
        elif db_type.value == 'sqlite':
            con = sqlite3.connect(**SQLITE_COF)
        self.__con = con
        self.__cursor = self.__con.cursor()

    def close_all(self):
        print("关闭游标和连接对象")
        self.__cursor.close()
        self.__con.close()
        self.__cursor = None
        self.__con = None

    def __del__(self):
        try:
            self.close_all()
        except Exception as e:
            pass

    def create_table(self, sql: str) -> None:
        self.__cursor.execute(sql)
        print("创建表成功")

    def drop_table(self, table_name: str) -> None:
        # 删除表要慎重
        # 1）表是否存在  2）表中的数据怎么办
        sql = f"drop table IF EXISTS {table_name}"
        self.__cursor.execute(sql)
        self.__con.commit()
        print("删除表成功")

    # lazyload=懒加载=用到的时候再创建
    def insert(self, sql: str, params: list = None) -> None:
        for param in params:
            self.__cursor.execute(sql, param)
            self.__con.commit()
            print("插入成功")

    def delete(self, sql: str) -> None:
        self.__cursor.execute(sql)
        self.__con.commit()

    def update(self, sql: str, param: tuple = None) -> None:
        self.__cursor.execute(sql, param)
        self.__con.commit()

    def fetchone(self, sql: str, param: tuple = None) -> tuple:
        self.__cursor.execute(sql, param)
        return self.__cursor.fetchone()

    def fetchall(self, sql: str, param: tuple = None) -> list:
        self.__cursor.execute(sql, param)
        return self.__cursor.fetchall()
